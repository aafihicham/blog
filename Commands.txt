Sélectionner des données (SELECT) :

SELECT * FROM users;

Ajouter des données (INSERT INTO) :

INSERT INTO users (first_name, second_name, email) VALUES ('John', 'Doe', 'john@example.com');

Mettre à jour des données (UPDATE) :

UPDATE users SET email = 'new_email@example.com' WHERE id = 1;

Supprimer des données (DELETE FROM) :

DELETE FROM users WHERE id = 1;
